🟠 Nível 3 — Lógica Booleana (Exercícios 11 a 14)
Ex 11Entrada no Evento (Operador E)
programa {
  // 11º Questão - Igor Frank Moreno
  funcao inicio() {
    inteiro idade
    cadeia resposta
    logico documento
    escreva("Qual é a sua idade?: ")
    leia(idade)
    escreva("Você tem documento? (S/N): ")
    leia(resposta)

    se (resposta == "S" ou resposta == "s") {
      documento = verdadeiro
      escreva("Você declarou que TEM documento! \n")
    } senao se (resposta == "N" ou resposta == "n") {
      documento = falso
      escreva("Você declarou que NÂO TEM documento! \n")
    } senao {
      escreva("Houve algo errado, finalizando sistema. \n")
    }

    se (idade >= 18 e documento == verdadeiro) {
      escreva("Entrada Liberada!")
    } senao {
      escreva("Houve algum erro... Entrada bloqueada!")
    }
  }
}

    enquanto (count <= 5) {
      escreva(count)
      count ++
    }
    escreva("\n")
    inteiro i
    para (i = 1; i <= 5; i++) {
      escreva(i)
    }
    escreva("\n")

    }
  }

Ex 16Soma Acumulada de 1 até 5
programa {
  // 16º Questão - Igor Frank Moreno
  funcao inicio() {
    inteiro count = 1
    inteiro total = 0

    enquanto (count <= 5) {
      escreva(count)
      total = total + count
      count ++
    }
    escreva("\n")
    escreva(total)
    escreva("\n")

    // Versão com para
    inteiro total1 = 0
    inteiro i
    para (i = 1; i <= 5; i++) {
      escreva(i)
      total1 = total1 + i
    }
    escreva("\n")
    escreva(total1)
    escreva("\n")

    // Versão com para declarando no próprio laço
    inteiro total2 = 0
    para (inteiro c = 1; c <= 5; c++) {
      escreva(c)
      total2 = total2 + c
    }
    escreva("\n")
    escreva(total2)
    escreva("\n")
  }
}
